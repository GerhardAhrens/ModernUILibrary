private static void MenuPoint1()
{
    ConsoleHelper.Clear();
    ConsoleHelper.Write($"\nBeenden mit 'x'");
    using (var spinner = new ConsoleSpinner(17, 1,delay:500))
    {
        spinner.Start();

        while (true)
        {
            ConsoleKey key = Console.ReadKey(true).Key;
            if (key == ConsoleKey.X)
            {
                spinner.Stop();
                break;
            }
        }
    }

    ConsoleHelper.Wait();
}
